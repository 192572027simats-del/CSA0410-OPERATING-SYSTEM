#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>

int main()
{
    int source, copy;
    char buffer[100];
    int bytes;

    source = open("source.txt", O_RDONLY);
    copy = open("copy.c", O_WRONLY | O_CREAT | O_TRUNC, 0644);

    while ((bytes = read(source, buffer, 100)) > 0)
    {
        write(copy, buffer, bytes);
    }

    close(source);
    close(copy);

    printf("File copied successfully.\n");

    return 0;
}
